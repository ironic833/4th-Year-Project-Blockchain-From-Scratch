/**

 *

/**
 * @memberOf ElectronWindow
 * Module dependencies
 */

const { app, BrowserWindow } = require("electron");
const server = require('./app');

/**
 * @memberOf  ElectronWindow
 * Global variables
 */

let mainWindow;

/**
 * Creates the main browser window.
 * @memberOf  ElectronWindow
 * @function createWindow
 */
function createWindow() {
  /**
   * Create a new browser window with the specified options.
   * @memberOf  ElectronWindow
   * @type {BrowserWindow}
   */
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 700,
    title: "Auction Chain",
    webPreferences: {
      nodeIntegration: true,
    },
    autoHideMenuBar: true,
    icon: 'client/src/assets/icon.png'
  });

  /**
   * @memberOf  ElectronWindow
   * Load the main window with the specified URL.
   */
  mainWindow.loadURL("http://localhost:3000");

  /**
   * @memberOf  ElectronWindow
   * Set the `mainWindow` variable to `null` when the window is closed.
   */
  mainWindow.on("closed", function () {
    mainWindow = null;
  });
}

/**
 * @memberOf  ElectronWindow
 * Event listener for when Electron has finished initializing.
 */
app.on("ready", createWindow);

/**
 * Event listener for when the user resizes the window.
 * @memberOf  ElectronWindow
 * @param {Event} e - The resize event.
 * @param {number} x - The new width of the window.
 * @param {number} y - The new height of the window.
 */
app.on("resize", function (e, x, y) {
  mainWindow.setSize(x, y);
});

/**
 * @memberOf  ElectronWindow
 * Event listener for when all windows are closed.
 */
app.on("window-all-closed", function () {
  if (process.platform !== "darwin") {
    app.quit();
  }
});

/**
 * @memberOf  ElectronWindow
 * Event listener for when the application is activated.
 */
app.on("activate", function () {
  if (mainWindow === null) {
    createWindow();
  }
});
