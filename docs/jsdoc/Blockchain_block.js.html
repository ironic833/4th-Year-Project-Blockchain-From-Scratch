/**
 * A data object is used in the creation of a block and is a set of transactions
 * @typedef {Transaction[]} data
 */

/**
 * Class representing a block.
 * @class
 */
class Block {
  /**
   * Creates a new block.
   *
   * @param {number} timestamp - The timestamp of the block creation.
   * @param {string} lastHash - The hash of the previous block in the chain.
   * @param {string} hash - The hash of the current block.
   * @param {data} data - The data to be stored in the block.
   * @param {number} nonce - The nonce value used in the proof-of-work algorithm.
   * @param {number} difficulty - The difficulty of the proof-of-work algorithm.
   */
  constructor({ timestamp, lastHash, hash, data, nonce, difficulty }) {
    this.timestamp = timestamp;
    this.lastHash = lastHash;
    this.hash = hash;
    this.data = data;
    this.nonce = nonce;
    this.difficulty = difficulty;
  }

  /**
   * Creates the genesis block of the blockchain.
   *
   * @returns {Block} The genesis block.
   */
  static genesis() {
    return new this(GENESIS_DATA);
  }

  /**
   * Mines a new block to add to the blockchain.
   *
   * @param {Block} lastBlock - The previous block in the blockchain.
   * @param {data} data - The data to be stored in the block.
   * @returns {Block} The newly mined block.
   */
  static mineBlock({ lastBlock, data }) {
    const lastHash = lastBlock.hash;
    let hash, timestamp;
    let { difficulty } = lastBlock;
    let nonce = 0;

    do {
      nonce++;
      timestamp = Date.now();
      difficulty = Block.adjustDifficulty({ originalBlock: lastBlock, timestamp });
      hash = cryptoHash(timestamp, lastHash, data, nonce, difficulty);
    } while (hexToBinary(hash).substring(0, difficulty) !== '0'.repeat(difficulty));

    return new this({ timestamp, lastHash, data, difficulty, nonce, hash });
  }

  /**
   * Adjusts the difficulty of the proof-of-work algorithm for mining the next block in the blockchain.
   *
   * @param {Block} originalBlock - The original block whose difficulty is being adjusted.
   * @param {number} timestamp - The timestamp of the new block being mined.
   * @returns {number} The adjusted difficulty.
   */
  static adjustDifficulty({ originalBlock, timestamp }) {
    const { difficulty } = originalBlock;

    if (difficulty < 1) return 1;

    if ((timestamp - originalBlock.timestamp) > MINE_RATE ) return difficulty - 1;

    return difficulty + 1;
  }
}

module.exports = Block;
