config.js

/**
 * @namespace config
 */

/**
 * The initial mine rate for the mining system.
 * @constant {number}
 */
const MINE_RATE = 1000;

/**
 * The initial proof of work difficulty level. This is used in the genesis block and the
 * ensuing proof of work operations will work off of this initial level.
 * @constant {number}
 */
const INITIAL_DIFFICULTY = 3;

/**
 * The genesis block data used in the creation of the chain's first block.
 * @constant {Object}
 * @property {number} timestamp - The timestamp of the genesis block.
 * @property {string} lastHash - The hash of the previous block in the chain.
 * @property {string} hash - The hash of the genesis block.
 * @property {number} difficulty - The proof of work difficulty level for the genesis block.
 * @property {number} nonce - The nonce value used in the proof of work algorithm for the genesis block.
 * @property {Array} data - The data stored in the genesis block.
 */
const GENESIS_DATA = {
    timestamp: 1,
    lastHash: '-----',
    hash: 'hash-one',
    difficulty: INITIAL_DIFFICULTY,
    nonce: 0,
    data: []
};

/**
 * The starting balance for all accounts in the blockchain.
 * @constant {number}
 */
const STARTING_BALANCE = 1000;

/**
 * The reward input object for the miner who is authorized to receive the mining reward.
 * @constant {Object}
 * @property {string} address - The address of the miner who is authorized to receive the mining reward.
 */
const REWARD_INPUT = {
    address: '*authorized-reward'
};

/**
 * The mining reward for the miner who successfully mines a new block.
 * @constant {number}
 */
const MINING_REWARD = 50;

/**
 * Exports the genesis block data, mine rate, starting balance, reward input, and mining reward as a module.
 * @module config
 * @type {Object}
 * @property {Object} GENESIS_DATA - The genesis block data.
 * @property {number} MINE_RATE - The mine rate for the mining system.
 * @property {number} STARTING_BALANCE - The starting balance for all accounts in the blockchain.
 * @property {Object} REWARD_INPUT - The reward input object for the miner who is authorized to receive the mining reward.
 * @property {number} MINING_REWARD - The mining reward for the miner who successfully mines a new block.
 */
module.exports = { GENESIS_DATA, MINE_RATE, STARTING_BALANCE, REWARD_INPUT, MINING_REWARD };
