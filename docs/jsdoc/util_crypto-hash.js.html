/**
 * Calculates the SHA256 hash of the inputs. May take several inputs
 *
 * @param {...any} inputs - The inputs to be hashed.
 * @returns {string} The resulting hash value in hexadecimal format.
 */
const cryptoHash = (...inputs) => {
  // Create a new SHA256 hash object
  const hash = crypto.createHash('sha256');

  // Convert each input to a string, sort them, and concatenate them with a space
  const sortedInputs = inputs.map(input => JSON.stringify(input)).sort().join(' ');

  // Update the hash object with the sorted input string
  hash.update(sortedInputs);

  // Return the hash value as a hexadecimal string
  return hash.digest('hex');
};

/**
 * Exports the cryptoHash class as a module.
 * @module cryptoHash
 */
module.exports = cryptoHash;
